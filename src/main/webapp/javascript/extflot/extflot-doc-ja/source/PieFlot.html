/**
 * @class Ext.ux.PieFlot
 * @extends Ext.ux.Flot
 * 円グラフ専用の Flot です。

 * 複数の系列をひとまとめにして、データごとに値をまとめて与えることができます。

data: [{
  label: ,
  data: [[, ], ...]
},{
...
}]


 *
 * data の X 値 (data[i][j][0]) には、その系列の ID を入れますが、
 * これは内部的には無視され、 Y 値 (data[i][j][1]) の和が
 * その系列の値になります。
 *
 * 対象の系列が円グラフで占める割合は、この各系列の値の比になります。
 * 

series: [
  { label: 'Niveau A', data: [[0,2]]},  # 40%
  { label: 'Niveau B', data: [[1,2]]},  # 40%
  { label: 'Niveau C', data: [[2,1]]},  # 20%
  { label: 'Niveau D', data: [[3,0]]}   #  0%
],


 *
 * 円グラフは操作の必要がないので、デフォルトのコンテキストメニューは
 * プロパティのみになっています。
 */
Ext.ux.PieFlot = Ext.extend(Ext.ux.Flot, {

  
/**
   * @cfg {Object} selection
   * mode が null になっています。
   */
  selection: { mode: null },

  
/**
   * @cfg {Bool/String/Object} tooltip
   * false になっています。
   */
  tooltip: false,

  
/**
   * @property contextMenu
   * @type Array
   * property のみになっています。
   */
  contextMenu: [{
    name: 'property',
    text: _('Property'),
    iconCls: 'puzzle',
    handler: function() { this.showProperty(); }
  }],

  
/**
   * @property datapointContextMenu
   * @type Array 
   * 無効になっています
   */
  datapointContextMenu: false,

	// private
	initComponent: function() {
		Ext.ux.PieFlot.superclass.initComponent.call(this);
	}
});
Ext.reg('pieflot', Ext.ux.PieFlot);
